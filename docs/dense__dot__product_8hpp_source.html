10// Force unrolling to avoid relying on optimizer decisions.
11// For example, older versions of GCC won't unroll at -O2, so we need to do it ourselves if we don't want a run-time nested loop.
12// Fortunately, compilers will stil auto-vectorize a manually-unrolled loop, so this won't be a pessimisation in the long term.
13template<std::size_t counter_ = 0, class Iterator1_, class Iterator2_, typename Output_, std::size_t accumulators_>
14void unrolled_dense_dot_product(const std::size_t idx, Iterator1_ start1, Iterator2_ start2, std::array<Output_, accumulators_>& dots) {
15 dots[counter_] += static_cast<Output_>(*(start1 + idx + counter_)) * static_cast<Output_>(*(start2 + idx + counter_));
16 if constexpr(counter_ + 1 < accumulators_) {
17 unrolled_dense_dot_product<counter_ + 1>(idx, start1, start2, dots);
18 }
19}

21template<std::size_t accumulators_, class Iterator1_, class Iterator2_, typename Output_>
22Output_ dense_dot_product(const std::size_t len, Iterator1_ start1, Iterator2_ start2, Output_ initial) {
23 if constexpr(accumulators_ == 1) {
24 Output_ dot = initial;
25 for (std::size_t i = 0; i < len; ++i) {
26 dot += static_cast<Output_>(*(start1 + i)) * static_cast<Output_>(*(start2 + i));
27 }
28 return dot;
29
30 } else {
31 const std::size_t cycles = len / accumulators_;
32 const std::size_t remainder = len % accumulators_;
33 std::array<Output_, accumulators_> dots{};
34
35 // One might think to initialize the array with the first cycle of the dot products, like:
36 //
37 // for (std::size_t i = 0; i < accumulators_; ++i) {
38 // dots[i] = *(start1 + i) * *(start2 + i);
39 // }
40 // counter = accumulators_;
41 //
42 // and setting c = 1 below, which saves us one set of additions.
43 // But when I test this out, this seems to be no better, and indeed, a slight pessimisation for len ~= 200.
44 // I would guess that the time saved by skipping the addition is offset by the increased size of the program.
45
46 for (std::size_t c = 0; c < cycles; ++c) {
47 // Product is fine as it is less than 'len', which must fit in a std::size_t.
48 unrolled_dense_dot_product(c * accumulators_, start1, start2, dots);
49 }
50
51 for (std::size_t i = 0; i < remainder; ++i) {
52 const auto idx = cycles * accumulators_ + i;
53 initial += *(start1 + idx) * *(start2 + idx);
54 }
55 return std::accumulate(dots.begin(), dots.end(), initial);
56 }
