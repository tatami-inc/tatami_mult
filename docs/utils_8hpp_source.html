18 const bool row,
19 const Index_ primary,
20 const Index_ secondary,

38 }
39 }

51template<typename Value_, typename Index_, class Zero_>
52FetchNonEmptySparseBlockInfo<Index_> fetch_non_empty_sparse_block(
53 tatami::OracularSparseExtractor<Value_, Index_>& ext,
54 std::vector<std::vector<Value_> >& vbuffers,
55 std::vector<std::vector<Index_> >& ibuffers,
56 std::vector<tatami::SparseRange<Value_, Index_> >& ranges,
57 std::vector<Index_>& non_empty,
58 Index_ position,
59 const Index_ length,
60 const int block_size,
61 Zero_ zero
62) {
63 non_empty.clear();
64 Index_ num_non_empty = 0;
65 bool all_non_empty = true;
66
67 // We assume that a check for remaining dimension elements was performed before continuing the blocked loop.
68 assert(position < length);
69
70 do {
71 auto lrange = ext.fetch(vbuffers[num_non_empty].data(), ibuffers[num_non_empty].data());
72 if (lrange.number == 0) {
73 zero(position);
74 all_non_empty = false;
75 ++position;
76 continue;
77 }
78
79 ranges[num_non_empty] = std::move(lrange);
80 non_empty.push_back(position);
81 ++num_non_empty;
82 ++position;
83
84 if (sanisizer::is_equal(num_non_empty, block_size)) {
85 break;
86 }
87 } while (position < length);
88
89 return FetchNonEmptySparseBlockInfo(position, num_non_empty, all_non_empty);

92template<typename Output_, typename DenseValue_, typename Value_, typename Index_>
93Output_ dense_sparse_dot_product(const DenseValue_* ptr, const tatami::SparseRange<Value_, Index_>& range) {
94 if (range.number == 0) {
95 return 0;
96 }
97
98 // Copying Eigen's use of two accumulators; effectively unrolls the loop a little for speed.
99 Output_ dot1 = 0, dot2 = 0;
100
101 Index_ s = 0;
102 const Index_ number_m1 = range.number - 1;
103 for (; s < number_m1; s += 2) {
104 dot1 += range.value[s] * ptr[range.index[s]];
105 dot2 += range.value[s + 1] * ptr[range.index[s + 1]];
106 }
107
108 if (s < range.number) {
109 dot1 += range.value[s] * ptr[range.index[s]];
110 }
111 return dot1 + dot2;
112}
