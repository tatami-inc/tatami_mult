18template<typename Value_, typename Index_, typename Right_, typename Output_>
19void dense_column_vector(const tatami::Matrix<Value_, Index_>& matrix, const Right_* rhs, Output_* output, int num_threads) {
20 const Index_ NR = matrix.nrow();
21 const Index_ NC = matrix.ncol();
22
23 std::optional<std::vector<std::optional<std::vector<Output_> > > > tmp_results;
24 const bool do_parallel = num_threads > 1;
25 if (do_parallel) {
26 tmp_results.emplace(sanisizer::cast<I<decltype(tmp_results->size())> >(num_threads - 1));
27 }
28 std::fill_n(output, NR, 0);
29
30 const auto num_used = tatami::parallelize([&](int t, Index_ start, Index_ length) -> void {
31 auto ext = tatami::consecutive_extractor<false>(matrix, false, start, length);
32 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NR);
33
34 Output_* optr;
35 std::optional<std::vector<Output_> > cur_output;
36 if (!do_parallel || t == 0) {
37 optr = output;
38 } else {
39 cur_output.emplace(tatami::cast_Index_to_container_size<std::vector<Output_> >(NR));
40 optr = cur_output->data();
41 }
42
43 for (Index_ c = 0; c < length; ++c) {
44 auto ptr = ext->fetch(buffer.data());
45 Output_ mult = rhs[start + c];
46 for (Index_ r = 0; r < NR; ++r) {
47 optr[r] += mult * ptr[r];
48 }
49 }
50
51 if (do_parallel && t > 0) {
52 (*tmp_results)[t - 1] = std::move(cur_output);
53 }
54 }, NC, num_threads);
55
56 if (do_parallel) {
57 for (int u = 1; u < num_used; ++u) {
58 const auto& tmp = *((*tmp_results)[u - 1]);
59 for (Index_ r = 0; r < NR; ++r) {
60 output[r] += tmp[r];
61 }
62 }
63 }
64}

321template<typename Value_, typename Index_, typename RightValue_, typename RightIndex_, typename Output_>
322void dense_column_tatami_sparse(
323 const tatami::Matrix<Value_, Index_>& matrix,
324 const tatami::Matrix<RightValue_, RightIndex_>& rhs,
325 const bool output_columnar,
326 Output_* const output,
327 int num_threads
328) {
329 const auto NR = matrix.nrow();
330 const auto NC = matrix.ncol();
331 assert(sanisizer::is_equal(NC, rhs.nrow()));
332 const auto num_rhs = rhs.ncol();
333
334 // The general strategy here is to drag the (hopefully small-ish) RHS matrix into memory,
335 // avoiding the need to repeatedly use tatami to iterate over it.
336 // We must have a row-major layout for this to work, we can't rely on blocking for a sparse RHS.
337
338 auto all_vbuffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightValue_> > >(NC);
339 auto all_ibuffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightIndex_> > >(NC);
340 auto all_ranges = tatami::create_container_of_Index_size<std::vector<tatami::SparseRange<RightValue_, RightIndex_> > >(NC);
341 populate_sparse_buffers(true, NC, num_rhs, rhs, all_vbuffers, all_ibuffers, all_ranges, num_threads);
342
343 const auto full_size_output = sanisizer::product_unsafe<std::size_t>(NR, num_rhs);
344 std::fill_n(output, full_size_output, 0);
345
346 const bool do_parallel = num_threads > 1;
347 std::optional<std::vector<std::optional<std::vector<Output_> > > > tmp_results;
348 if (do_parallel) {
349 tmp_results.emplace(sanisizer::cast<I<decltype(tmp_results->size())> >(num_threads - 1));
350 }
351
352 const auto num_used = tatami::parallelize([&](int t, Index_ start, Index_ length) -> void {
353 auto ext = tatami::consecutive_extractor<false>(matrix, false, start, length);
354 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NR);
355
356 Output_* optr;
357 std::optional<std::vector<Output_> > tmp_out;
358 if (do_parallel && t != 0) {
359 // If a full size array was allocated in 'output', we should be able to allocate a vector... hopefully.
360 tmp_out.emplace(sanisizer::cast<typename std::vector<std::vector<Output_> >::size_type>(full_size_output));
361 optr = tmp_out->data();
362 } else {
363 optr = output;
364 }
365
366 for (Index_ c = 0; c < length; ++c) {
367 const auto ptr = ext->fetch(buffer.data());
368 const auto& range = all_ranges[start + c];
369 if (output_columnar) {
370 for (I<decltype(range.number)> k = 0; k < range.number; ++k) {
371 const auto mult = range.value[k];
372 const auto outcol = range.index[k];
373 for (Index_ r = 0; r < NR; ++r) {
374 optr[sanisizer::nd_offset<std::size_t>(r, NR, outcol)] += mult * ptr[r];
375 }
376 }
377 } else {
378 for (Index_ r = 0; r < NR; ++r) {
379 const auto mult = ptr[r];
380 for (I<decltype(range.number)> k = 0; k < range.number; ++k) {
381 optr[sanisizer::nd_offset<std::size_t>(range.index[k], num_rhs, r)] += mult * range.value[k];
382 }
383 }
384 }
385 }
386
387 if (do_parallel && t > 0) {
388 (*tmp_results)[t - 1] = std::move(tmp_out);
389 }
390 }, NC, num_threads);
391
392 if (do_parallel) {
393 for (int u = 1; u < num_used; ++u) {
394 const auto& tmp = *((*tmp_results)[u - 1]);
395 for (std::size_t x = 0; x < full_size_output; ++x) {
396 output[x] += tmp[x];
397 }
398 }
399 }
400}
