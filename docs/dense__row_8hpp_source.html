50

55 // TODO: wouldn't hurt to use a blocking strategy here either, to reduce cache misses.
56 if (output_columnar) {

58 const auto rptr = rhs_ptrs[j];

60 optr[start + r] = std::inner_product(ptr, ptr + NC, rptr, static_cast<Output>(0));
61 }
62
63 } else {
64 auto&& optr = get_output(start + r);
65 for (I<decltype(num_rhs)> j = 0; j < num_rhs; ++j) {
66 const auto rptr = rhs_ptrs[j];
67 optr[j] = std::inner_product(ptr, ptr + NC, rptr, static_cast<Output>(0));
68 }
69 }
70
71 } else {
72 // TODO: collect values and use a block strategy here, lots of non-contiguous access in the hottest loop.

184template<typename Value_, typename Index_, typename RightValue_, typename RightIndex_, typename Output_>
185void dense_row_tatami_sparse(
186 const tatami::Matrix<Value_, Index_>& matrix,
187 const tatami::Matrix<RightValue_, RightIndex_>& rhs,
188 const bool output_columnar,
189 Output_* output,
190 int num_threads
191) {
192 const Index_ NR = matrix.nrow();
193 const Index_ NC = matrix.ncol();
194 assert(sanisizer::is_equal(NC, rhs.nrow()));
195 const auto num_rhs = rhs.ncol();
196
197 // The general strategy here is to drag the (hopefully small-ish) RHS matrix into memory,
198 // avoiding the need to repeatedly use tatami to iterate over it.
199 // For row-major RHS matrices, we forcibly extract columns if the output is also columnar.
200 // This avoids a suboptimal extraction within dense_row_sparse_vectors(), see the comments below.
201 // We consider a one-time suboptimal extraction to be cheaper than many suboptimal accesses for multiplication itself.
202
203 if (!rhs.prefer_rows() || output_columnar) {
204 auto all_vbuffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightValue_> > >(num_rhs);
205 auto all_ibuffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightIndex_> > >(num_rhs);
206 auto all_ranges = tatami::create_container_of_Index_size<std::vector<tatami::SparseRange<RightValue_, RightIndex_> > >(num_rhs);
207 populate_sparse_buffers(false, num_rhs, NC, rhs, all_vbuffers, all_ibuffers, all_ranges, num_threads);
208
209 if (output_columnar) {
210 tatami::parallelize([&](int, Index_ start, Index_ length) -> void {
211 auto ext = tatami::consecutive_extractor<false>(matrix, true, start, length);
212 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NC);
213 for (Index_ r = start, end = start + length; r < end; ++r) {
214 auto ptr = ext->fetch(buffer.data());
215 for (I<decltype(num_rhs)> j = 0; j < num_rhs; ++j) {
216 const auto& range = all_ranges[j];
217 // Non-contiguous writes are a little annoying and more susceptible to false sharing,
218 // but they're also relatively infrequent, so we'll just accept it.
219 output[sanisizer::nd_offset<std::size_t>(r, NR, j)] = dense_sparse_dot_product<Output_>(ptr, range);
220 }
221 }
222 }, NR, num_threads);
223
224 } else {
225 tatami::parallelize([&](int, Index_ start, Index_ length) -> void {
226 auto ext = tatami::consecutive_extractor<false>(matrix, true, start, length);
227 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NC);
228 for (Index_ r = start, end = start + length; r < end; ++r) {
229 auto ptr = ext->fetch(buffer.data());
230 for (I<decltype(num_rhs)> j = 0; j < num_rhs; ++j) {
231 const auto& range = all_ranges[j];
232 output[sanisizer::nd_offset<std::size_t>(j, num_rhs, r)] = dense_sparse_dot_product<Output_>(ptr, range);
233 }
234 }
235 }, NR, num_threads);
236 }
237
238 } else {
239 auto all_vbuffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightValue_> > >(NC);
240 auto all_ibuffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightIndex_> > >(NC);
241 auto all_ranges = tatami::create_container_of_Index_size<std::vector<tatami::SparseRange<RightValue_, RightIndex_> > >(NC);
242 populate_sparse_buffers(true, NC, num_rhs, rhs, all_vbuffers, all_ibuffers, all_ranges, num_threads);
243
244 /* For columnar output, the per-thread loop would look something like this:
245
246 for (Index_ r = start, end = start + length; r < end; ++r) {
247 auto ptr = ext->fetch(buffer.data());
248 for (Index_ c = 0; c < NC; ++c) {
249 auto range = get_rhs(c);
250 for (I<decltype(range.number)> x = 0; x < range.number; ++x) {
251 auto&& optr = get_output(range.index[x]);
252 optr[r] += ptr[range.index[x]] * range.value[x];
253 }
254 }
255 }
256
257 * The innermost loop is pretty gross given the non-contiguous access.
258 * There is also a high risk of false sharing in a multi-threaded context.
259 * We've got a sparse matrix so we can't easily use blocking; so, we'll avoid columnar output if we have a non-columnar RHS.
260 */
261 assert(!output_columnar);
262
263 tatami::parallelize([&](int, Index_ start, Index_ length) -> void {
264 auto ext = tatami::consecutive_extractor<false>(matrix, true, start, length);
265 auto buffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NC);
266 for (Index_ r = start, end = start + length; r < end; ++r) {
267 auto ptr = ext->fetch(buffer.data());
268 for (Index_ c = 0; c < NC; ++c) {
269 const auto& range = all_ranges[c];
270 for (I<decltype(range.number)> x = 0; x < range.number; ++x) {
271 output[sanisizer::nd_offset<std::size_t>(range.index[x], num_rhs, r)] += ptr[range.index[x]] * range.value[x];
272 }
273 }
274 }
275 }, NR, num_threads);
276 }
277}
