53void multiply_sparse_row_with_dense_row_matrix_to_column_output(
54 const tatami::Matrix<LeftValue_, LeftIndex_>& left,
55 const tatami::Matrix<RightValue_, RightIndex_>& right,
56 Output_* const output,
57 const MultiplySparseRowWithDenseRowMatrixToColumnOutputOptions& options
58) {
59 const auto left_NR = left.nrow();
60 const auto common_dim = left.ncol();
61 const auto right_NC = right.ncol();
62
63 auto right_buffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightValue_> > >(common_dim);
64 auto right_ptrs = tatami::create_container_of_Index_size<std::vector<const RightValue_*> >(common_dim);
65 populate_dense_buffers(true, common_dim, right_NC, right, right_buffers, right_ptrs, options.num_threads);
66
67 tatami::parallelize([&](int, LeftIndex_ start, LeftIndex_ length) -> void {
68 auto ext = tatami::consecutive_extractor<true>(left, true, start, length);
69 auto vbuffer = tatami::create_container_of_Index_size<std::vector<LeftValue_> >(common_dim);
70 auto ibuffer = tatami::create_container_of_Index_size<std::vector<LeftIndex_> >(common_dim);
71
72 auto tmp_output = tatami::create_container_of_Index_size<std::vector<Output_> >(right_NC);
73 std::vector<LeftIndex_> left_empty;
74
75 for (LeftIndex_ lr = 0; lr < length; ++lr) {
76 const auto range = ext->fetch(vbuffer.data(), ibuffer.data());
77 if (range.number == 0) {
78 left_empty.push_back(lr);
79 continue;
80 }
81
82 for (LeftIndex_ x = 0; x < range.number; ++x) {
83 const auto rightrow = right_ptrs[range.index[x]];
84 const auto mult = range.value[x];
85 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
86 tmp_output[rc] += mult * rightrow[rc];
87 }
88 }
89
90 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
91 output[sanisizer::nd_offset<std::size_t>(start + lr, left_NR, rc)] = tmp_output[rc];
92 tmp_output[rc] = 0;
93 }
94 }
95
96 if (left_empty.size()) {
97 // Zeroing the empty rows that we previously skipped. This is done with near-contiguous memory,
98 // so if there are many empty LHS rows, their special-casing should improve efficiency.
99 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
100 for (const auto lr : left_empty) {
101 output[sanisizer::nd_offset<std::size_t>(start + lr, left_NR, rc)] = 0;
102 }
103 }
104 }
105 }, left_NR, options.num_threads);
106}
