64void multiply_sparse_row_with_dense_row_matrix_to_column_output(
65 const tatami::Matrix<LeftValue_, LeftIndex_>& left,
66 const tatami::Matrix<RightValue_, RightIndex_>& right,
67 Output_* const output,
68 const MultiplySparseRowWithDenseRowMatrixToColumnOutputOptions& options
69) {
70 const auto left_NR = left.nrow();
71 const auto common_dim = left.ncol();
72 const auto right_NC = right.ncol();
73
74 auto right_buffers = tatami::create_container_of_Index_size<std::vector<std::vector<RightValue_> > >(common_dim);
75 auto right_ptrs = tatami::create_container_of_Index_size<std::vector<const RightValue_*> >(common_dim);
76 populate_dense_buffers(true, common_dim, right_NC, right, right_buffers, right_ptrs, options.num_threads);
77
78 tatami::parallelize([&](int, LeftIndex_ start, LeftIndex_ length) -> void {
79 auto ext = tatami::consecutive_extractor<true>(left, true, start, length);
80 auto vbuffer = tatami::create_container_of_Index_size<std::vector<LeftValue_> >(common_dim);
81 auto ibuffer = tatami::create_container_of_Index_size<std::vector<LeftIndex_> >(common_dim);
82
83 if (options.block_size == 1) {
84 auto tmp_output = tatami::create_container_of_Index_size<std::vector<Output_> >(right_NC);
85 std::vector<LeftIndex_> left_empty;
86
87 for (LeftIndex_ lr = 0; lr < length; ++lr) {
88 const auto range = ext->fetch(vbuffer.data(), ibuffer.data());
89 if (range.number == 0) {
90 left_empty.push_back(lr);
91 continue;
92 }
93
94 for (LeftIndex_ x = 0; x < range.number; ++x) {
95 const auto rightrow = right_ptrs[range.index[x]];
96 const Output_ mult = range.value[x];
97 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
98 tmp_output[rc] += mult * static_cast<Output_>(rightrow[rc]);
99 }
100 }
101
102 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
103 output[sanisizer::nd_offset<std::size_t>(start + lr, left_NR, rc)] = tmp_output[rc];
104 }
105 std::fill(tmp_output.begin(), tmp_output.end(), 0);
106 }
107
108 if (left_empty.size()) {
109 // Zeroing the empty rows that we previously skipped. This is done with near-contiguous memory,
110 // so if there are many empty LHS rows, their special-casing should improve efficiency.
111 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
112 for (const auto lr : left_empty) {
113 output[sanisizer::nd_offset<std::size_t>(start + lr, left_NR, rc)] = 0;
114 }
115 }
116 }
117
118 } else {
119 const auto max_block_rows = sanisizer::min(length, options.block_size);
120 std::vector<Output_> tmp_output(sanisizer::product<typename std::vector<Output_>::size_type>(max_block_rows, right_NC));
121
122 LeftIndex_ lr = 0;
123 while (lr < length) {
124 const LeftIndex_ lrnum = sanisizer::min(options.block_size, length - lr);
125 bool any_non_empty = false;
126 for (LeftIndex_ lrcopy = 0; lrcopy < lrnum; ++lrcopy) {
127 const auto range = ext->fetch(vbuffer.data(), ibuffer.data());
128 if (range.number == 0) {
129 continue;
130 }
131 any_non_empty = true;
132
133 for (LeftIndex_ x = 0; x < range.number; ++x) {
134 const auto rightrow = right_ptrs[range.index[x]];
135 const Output_ mult = range.value[x];
136 for (RightIndex_ rc = 0; rc < right_NC; ++rc) {
137 tmp_output[sanisizer::nd_offset<std::size_t>(rc, right_NC, lrcopy)] += mult * static_cast<Output_>(rightrow[rc]);
138 }
139 }
140 }
141
142 // Now doing a blocked transposition.
143 // This is, in fact, the only purpose of the blocking here.
144 // We do this even if there were no non-empty LHS rows, because we have to zero the output array anyway.
145 RightIndex_ rc = 0;
146 while (rc < right_NC) {
147 const RightIndex_ rcend = rc + sanisizer::min(options.block_size, right_NC - rc);
148 for (LeftIndex_ lrcopy = 0; lrcopy < lrnum; ++lrcopy) {
149 for (auto rcopy = rc; rcopy < rcend; ++rcopy) {
150 const auto val = tmp_output[sanisizer::nd_offset<std::size_t>(rcopy, right_NC, lrcopy)];
151 output[sanisizer::nd_offset<std::size_t>(start + lr + lrcopy, left_NR, rcopy)] = val;
152 }
153 }
154 rc = rcend;
155 }
156
157 // Too much effort to track individual non-empty rows, but if they're all empty, we skip the zeroing.
158 // If a few are non-empty, a single memset call is probably faster anyway than splitting it up.
159 if (any_non_empty) {
160 std::fill_n(tmp_output.begin(), sanisizer::product_unsafe<std::size_t>(right_NC, lrnum), 0);
161 }
162
163 lr += lrnum;
164 }
165 }
166 }, left_NR, options.num_threads);
167}
